Rede empresarial¶

Na construo da rede empresarial, a ideia fazer os seguintes passos:

  • Distribuir o nmero de empresas por tamanho da empresa seguindo uma distribuio na forma de uma lei de potncia prxima a de uma lei de Zipf, com parmetros escolhidos de acordo com a populao total e um nmero mximo de tamanho (ou mdio?).

  • Distribuir as empresas pelos blocos, com pesos de acordo com a populao de cada bloco.

  • Alocar os indivduos em cada empresa, levando em considerao uma determinada distncia mdia de deslocamento.

$$ \mathcal{P}(n > k) = f_3(k; a, c, k_\textrm{max}) = \frac{\displaystyle \left(\frac{\displaystyle 1 + \frac{k_\textrm{max}}{a}}{\displaystyle 1 + \frac{k}{a}}\right)^c - 1}{\displaystyle \left( 1 + \frac{k}{a}\right)^c -1}, \qquad 0 \leq k \leq k_\textrm{max} $$

def zipf3(a, c, k_max, k):
    '''
    retorna a frao de indivduos em empresas de tamanho maior que k.
    '''
    return (((1.0 + k_max/a)/(1.0 + k/a))**c - 1)/((1 + k_max/a)**c - 1.0)

def zipf3_sec(a, c, k_max, k0, k1):
    '''
    retorna a frao de individuos em empresas de tamanho maior ou igual a k0 e menor que k1.
    '''
    return zipf3(a, c, k_max, k0) - zipf3(a, c, k_max, k1)

def zipf3e(a, c, k_max, k):
    '''
    retorna a frao de empresas de tamanho k.
    '''
    return zipf3_sec(a, c, k_max, k-1, k)/k

def distribui_empresas_por_tamanho(num_pop, num_pop_emp, tam_0, z3_a, z3_c, z3_max, verbose=False):

    emp_tams_z3 = np.arange(tam_0, z3_max)
    emps_num_z3 = (num_pop_emp*zipf3e(z3_a, z3_c, z3_max, emp_tams_z3)).astype(int)
    emps_num_z3 = emps_num_z3[emps_num_z3>0]
    emp_tams_z3 = np.array(list(range(tam_0, tam_0 + len(emps_num_z3))))
    emps_pop_z3 = np.array([(tam_0 + k)*emps_num_z3[k] for k in range(len(emps_num_z3))])

    if not len(emp_tams_z3):
        print('No foi possvel distribuir as empresas, tente com outros parmetros')
    elif verbose:
        print(f'Total da populao: {num_pop}')
        print(f'Total da fora de trabalho: {num_pop_emp}')
        print(f'Nmero de tamanhos de empresas: {len(emps_num_z3)}')
        print(f'Nmero de empresas: {emps_num_z3.sum()}')
        print(f'Tamanhos de empresas: de {emp_tams_z3.min()} a {emp_tams_z3.max()}')
        print(f'Nmero de indivduos nas empresas: {emps_pop_z3.sum()}')
        print('Porcentagem de indivduos da fora de trabalho nas empresas: '
              + f'{100*emps_pop_z3.sum()/num_pop_emp:.1f}%')
        print(f'Distribuio de empresas por tamanho: \n{emps_num_z3}')
        print(f'Distribuio de indivduos por tamanho de empresa: \n{emps_pop_z3}')
    
    return emp_tams_z3, emps_num_z3, emps_pop_z3
